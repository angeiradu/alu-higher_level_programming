import unittest

class AddIntegerTest(unittest.TestCase):
    def test_add_integer_with_ints(self):
        self.assertEqual(add_integer(1, 2), 3)
        self.assertEqual(add_integer(0, -5), -5)

    def test_add_integer_with_floats(self):
        self.assertEqual(add_integer(0.5, 2.5), 3)
        self.assertEqual(add_integer(-1.5, 1.5), 0)

    def test_add_integer_with_default_value(self):
        self.assertEqual(add_integer(3), 101)

    def test_add_integer_with_invalid_arguments(self):
        with self.assertRaises(TypeError):
            add_integer("Hello", 2)
        with self.assertRaises(TypeError):
            add_integer(2, "World")

if __name__ == '__main__':
    unittest.main()
